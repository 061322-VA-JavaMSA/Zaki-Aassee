drop table if exists ers_user_roles;
create table ers_user_roles( 
ers_user_role_id int primary key,
user_role varchar (20)

);

drop table if exists  ers_users;

create table ers_users( 
ers_user_id int primary key,
ers_username varchar (50) unique not null, 
ers_password varchar (50) not null,
user_firstname varchar (50),
user_lastname varchar (50),
user_email varchar (50),
user_role_id int references ers_user_roles(ers_user_role_id)

);



drop table if exists ers_reimb_status;
create table ers_reimb_status( 
reimb_status_id int primary key,
reimb_status varchar (20)

);

drop table if exists ers_reimb_type;
create table ers_reimb_type (
reimb_type_id int primary key,
reimb_type varchar (20)


);

drop table if exists ers_main;
create table ers_main (
reimb_id int primary key, 
reimb_amount int, 
reimb_submitted timestamp,
reimb_resolved timestamp,
reimb_description varchar(300),
reimb_receipt bytea,
reimb_author int references ers_users(ers_user_id),
reimb_resolver int  references ers_users(ers_user_id),
reimb_status_id int references ers_reimb_status (reimb_status_id),
reimb_type_id int references ers_reimb_type(reimb_type_id)

);
